HOW IT WORKS:
───────────────

Step 1: User tries to access protected feature
Step 2: Code checks: if (!user)
Step 3: If not signed in:
        - Show toast: "Please sign in to..."
        - Redirect to /auth?tab=signin
Step 4: If signed in:
        - Navigation proceeds normally

CODE PATTERN:
──────────────

// Pattern 1: Link with onClick
const handleClick = (e: React.MouseEvent<HTMLAnchorElement>) => {
  if (!user) {
    e.preventDefault();
    toast.error("Please sign in...");
    navigate("/auth?tab=signin");
  }
};

// Pattern 2: Button with onClick
onClick={() => {
  if (!user) {
    toast.error("Please sign in...");
    navigate("/auth?tab=signin");
  } else {
    navigate("/target-page");
  }
}}
